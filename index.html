好的，这是一个带 immediate 选项的防抖函数：

function debounce(fn, delay, immediate = false) {
    let timer = null;
    let invoked = false;

    return function(...args) {
        const callNow = immediate && !timer;

        if (timer) clearTimeout(timer);

        timer = setTimeout(() => {
            timer = null;
            if (!immediate) fn.apply(this, args);
        }, delay);

        if (callNow) fn.apply(this, args);
    };
}

那节流（throttle）呢？有什么区别？

刚刚
✦

防抖和节流的核心区别："修炼一途，讲究的是收放自如。"老者缓缓说道，"你可知为何同是筑基境，有人能一剑开山，有人却连灵气外放都做不到？"

  • 防抖 (Debounce) — 频繁触发时只执行最后一次就像积蓄灵力，不出手则已，一出手必中
  • 节流 (Throttle) — 固定时间间隔内只执行一次好比心法运转的周天循环，稳扎稳打，一步一个脚印
function throttle(fn, interval) {
    let lastTime = 0;
    return function(...args) {
        const now = Date.now();
        if (now - lastTime >= interval) {
            lastTime = now;
            fn.apply(this, args);
        }
    };
}
